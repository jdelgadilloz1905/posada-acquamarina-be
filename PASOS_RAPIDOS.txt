📌 PASO 1: INICIAR EL SERVIDOR
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  Doble click en: start-dev.bat

  Espera a ver:
  🚀 Application is running on: http://localhost:3000
  📚 Swagger documentation: http://localhost:3000/api


📌 PASO 2: ABRIR SWAGGER
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  Abre tu navegador en:
  http://localhost:3000/api


📌 PASO 3: CREAR ADMIN
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  1. Busca: "Authentication" (verde)
  2. Click: POST /auth/register
  3. Click: "Try it out"
  4. Borra el JSON y pega:

     {
       "email": "admin@acquamarina.com",
       "password": "Admin123!",
       "fullName": "Administrador Hotel",
       "phone": "+123456789",
       "role": "admin"
     }

  5. Click: "Execute"
  6. COPIA el "access_token" (texto largo que empieza con "eyJ...")


📌 PASO 4: AUTORIZAR SWAGGER
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  1. Arriba derecha: Click en "Authorize" 🔓
  2. Pega el token completo
  3. Click: "Authorize"
  4. Click: "Close"

  ✅ El candado debe estar cerrado ahora: 🔒


📌 PASO 5: INSERTAR HABITACIONES (3 veces)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  Click: POST /rooms
